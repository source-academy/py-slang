\subsection*{Loops}

\subsubsection*{\texttt{\textbf{for}} loop through a range with one argument}

The expression in the following syntax should evaluate to an integer:
\[
  \textbf{\texttt{for}}\ \textit{name}\ \textbf{\texttt{in
  range(}}\ \textit{expression}\ \textbf{\texttt{)}}\ \textbf{\texttt{:}}\ \textit{block}
\]

This roughly stands for:
\begin{center}
  \begin{minipage}{90mm}
\begin{lstlisting}
_loop_i = 0
_end = ^$\textrm{\textit{expression}}$^

while _loop_i < _end:
    ^$\textrm{\textit{name}}$^ = _loop_i
    ^$\textrm{\textit{block}}$^
    _loop_i = _loop_i + 1
\end{lstlisting}
  \end{minipage}
\end{center}

\subsubsection*{\texttt{\textbf{for}} loop through a range with two arguments}

The expressions in the following syntax should both evaluate to integers:
\[
  \textbf{\texttt{for}}\ \textit{name}\ \textbf{\texttt{in
    range(}}\ \textit{expression}_1
  \textbf{\texttt{,}}\ \textit{expression}_2\ \textbf{\texttt{)}}\ \textbf{\texttt{:}}\ \textit{block}
\]

This roughly stands for:
\begin{center}
  \begin{minipage}{90mm}
\begin{lstlisting}
_loop_i = ^$\textrm{\textit{expression}}_1$^
_end = ^$\textrm{\textit{expression}}_2$^

while _loop_i < _end:
    ^$\textrm{\textit{name}}$^ = _loop_i
    ^$\textrm{\textit{block}}$^
    _loop_i = _loop_i + 1
\end{lstlisting}
  \end{minipage}
\end{center}

\subsubsection*{\texttt{\textbf{for}} loop through a range with three arguments}

The expressions in the following syntax should all evaluate to integers:
\[
  \textbf{\texttt{for}}\ \textit{name}\ \textbf{\texttt{in
    range(}}\ \textit{expression}_1
    \textbf{\texttt{,}}\ \textit{expression}_2
  \textbf{\texttt{,}}\ \textit{expression}_3\ \textbf{\texttt{)}}\ \textbf{\texttt{:}}\ \textit{block}
\]

This roughly stands for:
\begin{center}
  \begin{minipage}{90mm}
\begin{lstlisting}
_loop_i = ^$\textrm{\textit{expression}}_1$^
_end = ^$\textrm{\textit{expression}}_2$^
_step = ^$\textrm{\textit{expression}}_3$^

if _step > 0:
    while _loop_i < _end:
        ^$\textrm{\textit{name}}$^ = _loop_i
        ^$\textrm{\textit{block}}$^
        _loop_i = _loop_i + _step
else:
    while _loop_i > _end:
        ^$\textrm{\textit{name}}$^ = _loop_i
        ^$\textrm{\textit{block}}$^
        _loop_i = _loop_i + _step
\end{lstlisting}
  \end{minipage}
\end{center}
